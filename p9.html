//ENTER AN ARRAY FROM USER AND FIND K-th SMALLEST NUMBER
            #include<stdio.h>
            int main(){
                int size,i,j;
                printf("enter the size of your array");
                scanf("%d",&size);
                int arr[size];
                for( i=0; i<size; i++){
                    printf("enter value for %d element \n",i+1);
                    scanf("%d",&arr[i]);
                }
                for( i=0; i<size; i++){
                    for( j=i+1; j<size; j++){
                        if(arr[i] > arr[j]){
                            int temp = arr[i];
                            arr[i] = arr[j];
                            arr[j] = temp;
                        }
                    }
                }
                printf("enter the the kth smallest element  \n");
                int k;
                scanf("%d",&k);
                printf("%d",arr[k-1]);
                return 0;
            }
