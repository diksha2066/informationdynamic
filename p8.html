//ENTER AN ARRAY FROM USER AND FIND MAXIMUM AND MINIMUM VALUE
            #include<stdio.h>
            int main(){
                int size;
                printf("enter the size of your array");
                scanf("%d",&size);
                int arr[size],i;
                for(i=0; i<size; i++){
                    printf("enter value for %d element \n",i+1);
                    scanf("%d",&arr[i]);
                }
                int max = arr[0];
                for( i=0; i<size; i++){
                    if(arr[i] > max){
                        max = arr[i];
                    }
                }
                int min = arr[0];
                for(i=0; i<size; i++){
                    if(arr[i] < min){
                        min = arr[i];
                    }
                }
                printf("minimum value is %d \n",min);
                printf("maximum value is %d \n",max);
            
                return 0;
            }
