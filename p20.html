//ENTER A MATRIX OF 4 ROWS AND 2 COLUMNS FROM THE USER DISPLAY IT IN MATRIX FORM.
#include<stdio.h>
int main()
{
	int a[4][2],i,j;
	printf("enter the elements of matrix");
	for(i=0;i<4;i++)
	{
		for(j=0;j<2;j++)
		{
			scanf("%d",&a[i][j]);
		}
	}
	printf("MATRIX \n\t\n");
		for(i=0;i<4;i++)
	{
		for(j=0;j<2;j++)
		{
			printf("\t%d",a[i][j]);
		}
		printf("\t\n\t\n");
	}
	getch();
}
