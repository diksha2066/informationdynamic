//Print all the permutation of a string i.e. "ABCDE".
            #include<stdio.h>
            void swap(char *c1,char *c2){
            char temp;
            temp=*c1;
            *c1=*c2;
            *c2=temp;
            }
            void perm(char *s,int a,int n){
            int j;
            if(a==n){
            printf("%s\n",s);
            }
            else{
            for(j=a;j<=n;j++){
            swap((s+a),(s+j));
            perm(s,a+1,n);
            swap((s+a),(s+j));
            }
            }
            }
            
            void main(){
            char str[30];
            int n=0;
            printf("Enter a string: ");
            gets(str);
            printf("All posible permutations:\n");
            while(str[n]!='\0'){
            n++;
            }
            perm(str,0,n-1);
            getch();
            }
