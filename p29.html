//CONVERT A NUMBER INTO BINARY USING RECURSION
#include<stdio.h>
int decimalToBinary(int n)
{
    if (n == 0)
    {
        return 0;
    }
    else
    {
        return (n % 2 + 10 *decimalToBinary(n / 2));
    }
}
int main()
{
    int num;
    printf("Enter decimal number: ");
    scanf("%d", &num);
    printf ("binary number of %d is %d",num,decimalToBinary(num));
    return 0;
}
