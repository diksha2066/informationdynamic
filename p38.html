//ENTER THE STRING FROM USER AND CONVERT INTO TOGGLE CASE
            #include<stdio.h>
            void toggle_case(char str[], int length){
                if(str[0] < 'A' && str[0] > 'Z'){
                    str[0] = str[0] + 32;
                }
                int i;
                for( i=1; i<length; i++){
                    if(str[i] == ' '){
                        if(str[i+1] >= 'A' && str[i+1] <= 'Z'){
                            str[i+1] = str[i+1] + 32;
                        }
                    }
                    else if(str[i-1] != ' '){
                        if(str[i] >= 'a' && str[i] <= 'z'){
                            str[i] = str[i] - 32;
                        }
                    }
                }
            }
            
            int main(){
                char str[100];
                printf("enter string!");
                scanf("%[^\n]",&str);
                printf("string before togglecase %s \n",str);
                toggle_case(str,100);
                printf("string after toggle case %s \n",str);
                return 0;
            }
