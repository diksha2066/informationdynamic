//FIND THE SMALLEST ELEMENT IN THE INTEGER ARRAY USING RECURSION.
            #include <stdio.h>
            int find_minimum(int a[], int n) {
              int i, min, index;
            
              min = a[0];
              index = 0;
            
              for (i = 1; i< n; i++) {
                if (a[i] < min) {
                   index = i;
                   min = a[i];
                }
              }
            
              return index;
            }
             
            
            int main() {
              int i, array[100], size, location, minimum;
            
              printf("enter the size of the array\n");
              scanf("%d", &size);
            
              printf("Enter the elements of array\n");
            
              for (i= 0; i< size; i++)
                scanf("%d", &array[i]);
            
              location = find_minimum(array, size);
              minimum  = array[location];
            
              printf("smallest element is value = %d.\n",  minimum);
              return 0;
            }
