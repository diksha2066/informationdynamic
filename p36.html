//ENTER THE STRING FROM USER AND CONVERT INTO SENTENCE CASE
            #include<stdio.h>
            void tosentencecase(char str[], int length){
                    if(str[0] >= 'a' && str[0] <= 'z'){
                        str[0] = str[0] - 32;
                    }
                    int i;
                    for( i=0; i<length; i++){
                        if(str[i] == '.'){
                            str[i+2] = str[i+2] - 32;
                        }
                    }
                }
            int main(){
                int length = 100;
                printf("enter a string ! \n");
                char str[length];
                scanf("%[^\n]",str);
                printf("string before sentence case %s \n",str);
                tosentencecase(str, length);
                printf("string after sentence case %s \n",str);
                return 0;
            }
