//ENTER THE STRING FROM USER AND CONVERT IN LOWERCASE
#include<stdio.h>
void tolowercase(char str[], int length){
    	int i;
        for(i=0; i<length; i++){
            if(str[i] >= 'A' && str[i] <= 'Z'){
                str[i] = str[i] + 32;
            }
        }
    }

int main(){
    int length;
    printf("enter length of string !");
    scanf("%d",&length);
    printf("enter a string ! \n");
    char str[length];
    scanf("%s",&str);
    printf("string before lower case %s \n",str);
    tolowercase(str, length);
    printf("string after lower case %s \n",str);
    return 0;
}
