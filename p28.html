//CALCULATE THE SUM OF NATURAL NUMBER TILL 10 USING RECURSION
            #include <stdio.h>
            int addNumbers(int n)
             {
              if (n != 0)
                return n + addNumbers(n - 1);
              else
                return n;
            }
            
            int main() {
              printf("Sum of first ten natural number is  = %d", addNumbers(10));
              return 0;
            }

OUTPUT


        Sum of first ten natural number is  = 55
