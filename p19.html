//FIND THE LOCAL MAXIMA IN ARRAY
            #include<stdio.h>
            int main(){
                int size,i;
                printf("enter the size of your array");
                scanf("%d",&size);
                int arr[size];
                for( i=0; i<size; i++){
                    printf("enter value for %d element \n",i+1);
                    scanf("%d",&arr[i]);
                }
                
                for(i=1; i<size-2; i++){
                    if(arr[i] < arr[i+1] && arr[i] < arr[i-1]){
                        printf("local maxima is %d \n",arr[i]);
                    }
                }
                return 0; 
            }
