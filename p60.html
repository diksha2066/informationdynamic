#include <stdio.h> 
            #include <stdlib.h> 
            
            
            struct stack
            {
                char ch;
                struct stack *next;
            };
            
            struct stack *top = NULL, *temp, *newnode;
            
            int push(int count)
            {
                if (top == NULL)
                {
                    newnode = (struct stack *)malloc(sizeof(struct stack));
                    printf("Enter the character\n");
                    scanf(" %c", &(newnode->ch)); // Use a space before %c to skip newline character
                    newnode->next = NULL;
                    top = newnode;
                    count++;
                }
                else
                {
                    newnode = (struct stack *)malloc(sizeof(struct stack));
                    printf("Enter the character\n");
                    scanf(" %c", &(newnode->ch));
                    newnode->next = top;
                    top = newnode;
                    count++;
                }
                return count;
            }
            
            int pop(int count)
            {
                if (top == NULL)
                {
                    printf("Stack is empty\n");
                }
                else
                {
                    temp = top;
                    printf("Deleted data on the stack is %c\n", temp->ch);
                    top = top->next;
                    free(temp);
                    count--;
                }
                return count;
            }
            
            void peek()
            {
                if (top == NULL)
                {
                    printf("Stack is empty\n");
                }
                else
                {
                    printf("Data from the top of the stack is %c\n", top->ch);
                }
            }
            
            void isempty()
            {
                if (top == NULL)
                {
                    printf("Stack is empty\n");
                }
                else
                {
                    printf("Stack is not empty\n");
                }
            }
            
            void isfull()
            {
                newnode = (struct stack *)malloc(sizeof(struct stack));
                if (newnode == NULL)
                {
                    printf("Stack is full\n");
                }
                else
                {
                    printf("Stack is not full\n");
                }
                free(newnode);
            }
            
            int main(void)
            {
                int choice, count = 0;
                while (1)
                {
                    printf("\n1. Push\n");
                    printf("2. Pop\n");
                    printf("3. Peek\n");
                    printf("4. Isempty\n");
                    printf("5. Isfull\n");
                    printf("6. Exit\n");
                    printf("Enter your choice\n");
                    scanf("%d", &choice);
                    switch (choice)
                    {
                    case 1:
                        count = push(count);
                        break;
                    case 2:
                        count = pop(count);
                        break;
                    case 3:
                        peek();
                        break;
                    case 4:
                        isempty();
                        break;
                    case 5:
                        isfull();
                        break;
                    case 6:
                        while (top != NULL)
                        {
                            temp = top;
                            top = top->next;
                            free(temp);
                        }
                        exit(0);
                    default:
                        printf("Enter a valid number\n");
                    }
                }
            }
