//ENTER THE STRING FROM USER AND DISPLAY LENGTH OF THE STRING
            #include<stdio.h>
            int total_length(char str[]){
                int count = 0;
                int i;
                for( i=0; str[i] != '\0'; i++){
                    count++;
                }
                return count++;
            }
            
            int main(){
                char str[100];
                printf("enter string!");
                scanf("%[^\n]",&str);
                printf("your entered string is %s \n",str);
                printf(" length of string %d",total_length(str));
                return 0;
            }
