//ENTER AN ARRAY FROM USER AND PRINT ALL THE POSSIBLE COMBINATIONS OF ELEMENTS
            #include<stdio.h>
            int main(){
                int size,i,j,k;
                printf("enter the size of your array");
                scanf("%d",&size);
                int array[size];
                for( i=0; i<size; i++){
                    printf("enter value for %d element \n",i+1);
                    scanf("%d",&array[i]);
                }
                for( i=0; i<size; i++){
                    for( j=i; j<size; j++){
                        for( k=i; k<=j; k++){
                            printf("%d , ",array[k]);
                        }
                        printf("\n");
                    }
                }
                return 0;
            }
