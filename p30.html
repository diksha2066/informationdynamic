//FIND THE LARGEST ELEMENT IN THE INTEGER ARRAY USING RECURSION.
            #include <stdio.h>
            int find_maximum(int a[], int n) {
              int i, max, index;
            
              max = a[0];
              index = 0;
            
              for (i = 1; i< n; i++) {
                if (a[i] > max) {
                   index = i;
                   max = a[i];
                }
              }
            
              return index;
            }
             
            
            int main() {
              int i, array[100], size, location, maximum;
            
              printf("enter the size of the array\n");
              scanf("%d", &size);
            
              printf("Enter the elements of array\n");
            
              for (i= 0; i< size; i++)
                scanf("%d", &array[i]);
            
              location = find_maximum(array, size);
              maximum  = array[location];
            
              printf("largest element is value = %d.\n",  maximum);
              return 0;
            }
