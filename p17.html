                 i=0, j=0, k=0;

                    if(array1[i] > array2[j]){
                        arr3[k] = array2[j];
                        j++;
                        k++;
                    }

                        arr3[k] = array1[i];
                        i++;
                        k++;
                    }

                        arr3[k] = array1[i];
                        k++;

                        j++;
                        k++;
                    }
                    else if(j == size2){
                        arr3[k] = array1[i];
                        i++;
                        k++;
                    }
                }
                for(i=0; i<size1+size2; i++){
                    printf("%d ",arr3[i]);
                }
                return 0;
            }
