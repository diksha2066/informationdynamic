//ENTER AN ARRAY FROM USER AND COPY IT INTO ANOTHER ARRAY IN REVERSE ORDER
            #include<stdio.h>
            int main(){
                int size,i;
                printf("enter the size of your array");
                scanf("%d",&size);
                int arr[size];
                for( i=0; i<size; i++){
                    printf("enter value for %d element \n",i+1);
                    scanf("%d",&arr[i]);
                }
                int arr2[size];
                int j = size-1;
                for( i=0; i<=size; i++){
                    arr2[i] = arr[j];
                    j--;
                }
                for(i=0; i<size; i++){
                    printf("value at %d index of array2 is %d \n",i,arr2[i]);
                }
            
                return 0;
            }
