//ENTER AN ARRAY FROM USER AND FIND DUPLICATE VALUES
            #include<stdio.h>
            int main(){
                int size,i,j;
                printf("enter the size of your array");
                scanf("%d",&size);
                int array[size];
                for(i=0; i<size; i++){
                    printf("enter value for %d element \n",i+1);
                    scanf("%d",&array[i]);
                }
                for( i=0; i<size; i++){
                    for(j=i+1; j<size; j++){
                        if(array[i] == array[j]){
                            printf("duplicate values are %d ",array[i]);
                            printf("\n");
                        }
                    }
            }
                return 0;
            }
