//Enter a sentence from the user and then enter the word and count the occurence of that word in the sentence.
            #include<stdio.h>
            #include<string.h>
            void main(){
            char sentence[50];
            char word[20];
            int count=0,i,j,match,s_len,w_len;
            printf("Enter a sentence: ");
            gets(sentence);
            printf("Enter a word to count: ");
            gets(word);
            s_len=strlen(sentence);
            w_len=strlen(word);
            for(i=0;i<s_len-w_len+1;i++){
            match=1;
            for(j=0;j<w_len;j++){
            if(sentence[i+j]!=word[j]){
            match=0;
            break;
            }
            }
            if(match){
            sentence[i+j]=word[j];
            }
            }
            printf("The word '%s' appears %d times in the sentence.\n",word,count);
            getch();
            }
