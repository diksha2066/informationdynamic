#include <stdio.h> 
            void selection(int ar[], int size) {
                int i, j, temp;
                for (i = 0; i < size; i++) {
                    for (j = i; j < size; j++) {
                        if (ar[i] > ar[j]) {
                            temp = ar[i];
                            ar[i] = ar[j];
                            ar[j] = temp;
                        }
                    }
                }
            }
            
            int main() {
                int ar[10], i;
            
                printf("Enter the arrays\n");
                for (i = 0; i < 10; i++) {
                    printf("Enter the array at location %d: ", i + 1);
                    scanf("%d", &ar[i]);
                }
            
                selection(ar, 10);
            
                printf("Sorted Arrays are\n");
                for (i = 0; i < 10; i++) {
                    printf("%d\n", ar[i]);
                }
            
                return 0;
            }
