//CALCULATE FACTORIAL OF A NUMBER USING RECURSION 
#include<stdio.h>
int fact (int m)
{
if(m!=0)
return m*fact(m-1); 
else
return 1;

}
int main()
{
   int number, result;
       printf("Please enter a non-negative number or integer :");
       scanf("%d",&number);
       result= fact(number); 
       printf("The result of the factorial would be equal to : %d\n",result);

return 0;

}

OUTPUT


        Please enter a non-negative number or integer :5
        The result of the factorial would be equal to : 120
