//Enter the string from the user and count the number of characters,words,spaces and lines.
            #include<stdio.h>
            void main(){
            char str[100];
            int i=0,chr=0,wrd=1,spc=0,ln=0;
            printf("Enter a string: ");
            gets(str);
            for(i=0;str[i]!='\0';i++){
            chr++;
            if(str[i]==' '){
            wrd++;
            }
            if(str[i]==' '){
            spc++;
            }
            if(str[i]=='.'){
            ln++;
            }
            }
            printf("Character: %d",chr);
            printf("\nWord: %d",wrd);
            printf("\nSpace: %d",spc);
            printf("\nLine: %d",ln);
            getch();
            }

OUTPUT

        Character: 14
        Word: 2
        Space: 1
        Line: 0
