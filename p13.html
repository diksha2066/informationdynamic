//ENTER AN ARRAY FROM USER AND COUNT THE DUPLICATE VALUES
            #include<stdio.h>
            int main(){
                int size,i,j;
                printf("enter the size of your array");
                scanf("%d",&size);
                int arr[size];
                for(i=0; i<size; i++){
                    printf("enter value for %d element \n",i+1);
                    scanf("%d",&arr[i]);
                }
                int count = 0;
                for( i=0; i<size; i++){
                    for( j=i+1; j<size; j++){
                        if(arr[i] == arr[j]){
                            count++;
                        }
                    }
                }
                printf("number of duplicates are %d",count);
                return 0;
            }
