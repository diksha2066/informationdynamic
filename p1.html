//ENTER AN ARRAY FROM USER AND DISPLAY ALL THE VALUES
#include<stdio.h>
int main(){
    int size,i;
    printf("enter the size of your array");
    scanf("%d",&size);
    int array[size];
    for(i=0; i<size; i++) {
        printf("enter value for %d element \n",i+1);
        scanf("%d",&array[i]);
    }
    for(i=0; i<size; i++){
        printf("value of your array at %d index is %d \n",i,array[i]);
    }

    return 0;
}
