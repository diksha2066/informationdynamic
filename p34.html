            #include<stdio.h>
            void touppercase(char str[], int length){
                int i;
                    for( i=0; i<length; i++){
                        if(str[i] >= 'a' && str[i] <= 'z'){
                            str[i] = str[i] - 32;
                        }
                    }
                }
            int main(){
                int length;
                printf("enter length of string ");
                scanf("%d",&length);
                printf("enter a string ! \n");
                char str[length];
                scanf("%s",&str);
                printf("string before upper case %s \n",str);
                touppercase(str, length);
                printf("string after upper case %s \n",str);
                return 0;
            }
