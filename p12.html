//ENTER AN ARRAY FROM USER AND REMOVE THE DUPLICATE VALUES AND FILL WITH ZERO
            #include<stdio.h>
            int main(){
                int size,i,j;
                printf("enter the size of your array");
                scanf("%d",&size);
                int arr[size];
                for(i=0; i<size; i++){
                    printf("enter value for %d element \n",i+1);
                    scanf("%d",&arr[i]);
                }
                for( i=0; i<size; i++){
                    for( j=i+1; j<size; j++){
                        if(arr[i] == arr[j]){
                            printf("%d",arr[i]);
                            printf("\n");
                        }
                    }
                }
                for( i=0; i<size; i++){
                    for( j=i+1; j<size; j++){
                        if(arr[i] == arr[j]){
                            arr[i] = 0;
                            arr[j] = 0;
                        }
                    }
                }
                printf("after removing duplicate \n");
                for( i=0; i<size; i++){
                    printf("%d ",arr[i]);
                }
                return 0;
            }
