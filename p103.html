#include <stdio.h>  
            #include <string.h>  
            
            
            int bubble(char str[][10], int size) {
                int step = 0, i, j;
                char temp[10];
            
                for (i = 0; i < size; i++) {
                    for (j = 0; j < size - i; j++) {
                        if (strcmp(str[j], str[j + 1]) > 0) {
                            strcpy(temp, str[j]);
                            strcpy(str[j], str[j + 1]);
                            strcpy(str[j + 1], temp);
                        }
                    }
                }
                return step;
            }
            
            int main() {
