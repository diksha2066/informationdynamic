//ENTER AN ARRAY FROM USER AND COPY THE ARRAY INTO SECOND ARRAY
            #include<stdio.h>
            int main(){
                int size,i;
                printf("enter the size of your array");
                scanf("%d",&size);
                int arr[size];
                for( i=0; i<size; i++){
                    printf("enter value for %d element \n",i+1);
                    scanf("%d",&arr[i]);
                }
                int arr2[size];
                for( i=0; i<size; i++){
                    arr2[i] = arr[i];
                }
                printf("ARRAY IS COPIED \n");
                for( i=0; i<size; i++){
                    printf("value at %d index of array2 is %d \n",i,arr2[i]);
                }
            
                return 0;
            }
