//ENTER THE STRING FROM USER AND CONVERT INTO TITLE CASE
            #include<stdio.h>
            void totitle(char str[], int length){
                str[0] = str[0] - 32;
                int i ;
                for( i=0; i<length; i++){
                    if(str[i] == ' '){
                        str[i+1] = str[i+1] - 32;
                    }
                }
            }
            
            int main(){
                int length = 100;
                printf("enter a string ! \n");
                char str[length];
                scanf("%[^\n]",str);
                printf("string before title case %s \n",str);
                totitle(str,length);
                printf("string after title case %s \n",str);
                return 0;
            }
